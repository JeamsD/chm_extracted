Represents line styles.

Header file code

struct eDLineStyle

{

enum Style

{

eDSolidLine,

eDDottedLine,

eDCenterLine,

eDPhantomLine,

eDDashedLine,

eDDashDotLine,

eDHiddenLine,

eDLongDashLine

};

 

Style style;

 

float width;

 

eDLineStyle( Style s = eDSolidLine, float w = 0.0f )

: style( s ), width( w ) {}

};
